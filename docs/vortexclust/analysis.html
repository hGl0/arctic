 8    r"""
 9    Computes the within-cluster dispersion for evaluating clustering compactness.
10
11    :param X: Input data.
12    :type X: numpy.ndarray
13    :param labels: Cluster labels for each sample.
14    :type labels: numpy.ndarray
15
16    :raises ValueError: If `X` and `labels` have mismatched dimensions.
17    :raises TypeError: If `X` or `labels` is not a numpy array or pandas DataFrame.
18
19    :return: float representing within-cluster dispersion value.
20    :rtype: float
21    """
22    if not isinstance(X, (np.ndarray, pd.DataFrame)):
23        raise TypeError(f"X must be a NumPy array or pandas DataFrame, got {type(X)}")
24    if not isinstance(labels, (np.ndarray, pd.Series)):
25        raise TypeError(f"labels must be a NumPy array or pandas Series, got {type(labels)}")
26
27    X = np.asarray(X)
28    labels = np.asarray(labels)
29
30    # check dimensions
31    if len(X) != len(labels):
32        raise ValueError("Mismatched dimensions of input data 'X' and 'labels'.")
33
34    Wk = 0
35    unique_clusters = np.unique(labels)
36
37    for cluster in unique_clusters:
38        cluster_points = X[labels == cluster]
39        n_m = len(cluster_points)
40
41        # at least 2 points needed
42        if n_m > 1:
43            D_m = np.sum(pdist(cluster_points, metric='sqeuclidean'))  # euclidean used in comparative gap statistic
44            Wk += D_m / (2 * n_m)
45    return Wk
